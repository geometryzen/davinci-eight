Interface FacetVisitor

A facet visitor implementation is able to accept a value to be assigned to a WebGL uniform in a WebGL program. This interface is normally only implemented by a Material.

interface FacetVisitor {
    activeTexture(unit): void;
    matrix2fv(name, mat2, transpose): void;
    matrix3fv(name, mat3, transpose): void;
    matrix4fv(name, mat4, transpose): void;
    uniform1f(name, x): void;
    uniform1fv(name, data, srcOffset, srcLength): void;
    uniform1i(name, x): void;
    uniform1iv(name, data, srcOffset?, srcLength?): void;
    uniform2f(name, x, y): void;
    uniform2fv(name, data, srcOffset?, srcLength?): void;
    uniform2i(name, x, y): void;
    uniform2iv(name, data, srcOffset?, srcLength?): void;
    uniform3f(name, x, y, z): void;
    uniform3fv(name, data, srcOffset?, srcLength?): void;
    uniform3i(name, x, y, z): void;
    uniform3iv(name, data, srcOffset?, srcLength?): void;
    uniform4f(name, x, y, z, w): void;
    uniform4fv(name, data, srcOffset?, srcLength?): void;
    uniform4i(name, x, y, z, w): void;
    uniform4iv(name, data, srcOffset?, srcLength?): void;
}
